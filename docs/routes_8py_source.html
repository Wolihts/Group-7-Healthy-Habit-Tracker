152def diet_tips(diet_data,goal_data):
153 """
154 Takes in Diet and Goal Data and returns a list of tips
155 based on calculations made between the diet data and the goal data
156
157 Args:
158 diet_data (list): A list containing the diet data.
159 goal_data (list): A list containing the goal data
160
161 Returns:
162 list: A list of tips to be passed into the template and displayed on the webpage
163 """
164 tips=[]
165 if diet_data == [] or goal_data == []:
166 tips.append(f"Not enough data yet...")
167 return tips
168 diets = 0
169 diet_avg = 0
170 newest = diet_data[0][4]
171 goal = goal_data[0][4]
172 for diet in diet_data:
173 diet_avg += diet[4]
174 diets += 1
175 diet_avg = diet_avg / diets
176 if diet_avg >= goal:
177 tips.append(f"Your Average ({diet_avg:.2f}) currently meets or is passing your goal!")
178 else:
179 tips.append(f"Your Average ({diet_avg:.2f}) is currently below your goal!")
180 if newest >= goal:
181 tips.append(f"Your newest ({newest}) is above your goal! Keep it up!")
182 else:
183 tips.append(f"Your newest ({newest}) is below your goal! Don't let this become a trend.")
184 return tips

225 """
